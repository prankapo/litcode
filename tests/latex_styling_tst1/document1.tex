\noindent Using \tt vcode \rm environment:

\begin{vcode}
/* vcode Env */

    fprintf(stdout, "Hello, world\n");
    return 0;

\end{vcode}

\noindent Using \tt code \rm environment:

\begin{code}[Hello world in C]

    return 0;

\end{code}
